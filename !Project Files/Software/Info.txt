SPI Work-Around:
// Example: create a second SPI instance (VSPI) and use SD on it
#include <SPI.h>
#include <SD.h>

// pick free pins on your Heltec board (change to pins you have available)
#define SDCARD_MOSI 34 #Brown Wire
#define SDCARD_MISO 33 #Grey Wire
#define SDCARD_SCK  35 #White Wire
#define SDCARD_CS   36 #Yellow Wire

SPIClass SPI_2(VSPI);                 // or HSPI as needed
// initialize with (sck, miso, mosi, ss)
void setup(){
  SPI_2.begin(SDCARD_SCK, SDCARD_MISO, SDCARD_MOSI, SDCARD_CS);
  if (!SD.begin(SDCARD_CS, SPI_2)) {
    Serial.println("SD begin failed");
  } else {
    Serial.println("SD OK");
  }
}
